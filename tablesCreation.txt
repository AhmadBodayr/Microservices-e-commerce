CREATE TABLE IF NOT EXISTS session (
    user_id text PRIMARY KEY,
    name text
);

CREATE TABLE IF NOT EXISTS cart_item (
    item_id text,
    user_id text,
    sku text,
    small_mid_large_one_size text,
    color text,
    name text,
    size_number int,
    item_price double,
    quantity int,
    PRIMARY KEY (user_id)
);
